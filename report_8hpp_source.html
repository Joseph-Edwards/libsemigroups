156  template <typename TThisType, typename... TParams>
157  void operator()(TThisType const* ptr, TParams... args) {
158  if (_report) {
159  size_t tid = thread_id(std::this_thread::get_id());
160  std::lock_guard<std::mutex> lg(_mtx);
161  ColourGuard cg(tid);
162  std::cout << cg << "#" << tid << ": " << class_name(ptr) << ": ";
163  int dummy[sizeof...(TParams)] = {(std::cout << args, 0)...};
164  (void) dummy;
165  std::cout << '\n';
166  }

178  void reset_thread_ids() {
179  // Only do this from the main thread
180  LIBSEMIGROUPS_ASSERT(thread_id(std::this_thread::get_id()) == 0);
181  // Delete all thread_ids
182  _thread_map.clear();
183  _next_tid = 0;
184  // Reinsert the main thread's id
185  thread_id(std::this_thread::get_id());
186  }

188  size_t thread_id(std::thread::id tid) {
189  std::lock_guard<std::mutex> lg(_mtx);
190  auto it = _thread_map.find(tid);
191  if (it != _thread_map.end()) {
192  return (*it).second;
193  } else {
194  // Don't check the assert below because on a single thread machine
195  // (such as those used by appveyor), for an fp-semigroup more than 1
196  // thread will be used, and this assertion will fail.
197  // LIBSEMIGROUPS_ASSERT(_next_tid <=
198  // std::thread::hardware_concurrency());
199  _thread_map.emplace(tid, _next_tid++);
200  return _next_tid - 1;
201  }
202  }

204  private:
205  template <class T>
206  std::string class_name(T const* o) {
207  auto it = _class_name_map.find(typeid(*o).hash_code());
208  if (it != _class_name_map.end()) {
209  return (*it).second;
210  }
211  int status;
212  char* ptr = abi::__cxa_demangle(typeid(*o).name(), 0, 0, &status);
213  if (status == 0) { // successfully demangled
214  std::string full = std::string(ptr);
215  size_t last = full.size();
216  if (full.back() == '>') {
217  size_t bracket_count = 0;
218  do {
219  last = full.find_last_of("<>", last - 1);
220  if (last != std::string::npos) {
221  if (full.at(last) == '>') {
222  bracket_count++;
223  } else if (full.at(last) == '<') {
224  bracket_count--;
225  }
226  }
227  } while (bracket_count != 0);
228  }
229  size_t first = full.rfind("::", last - 1);
230  first = (first == std::string::npos ? 0 : first + 2);
231  free(ptr);
232  std::string out = full.substr(first, last - first);
233  _class_name_map.emplace(typeid(*o).hash_code(), out);
234  return out;
235  }
236  free(ptr);
237  std::string out;
238  _class_name_map.emplace(typeid(*o).hash_code(), out);
239  return out;
240  }
